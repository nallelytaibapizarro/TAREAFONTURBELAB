modelos_glmm <- list()  # guardamos los modelos para DHARMa después

for (tipo in levels(sens_long$sens_type)) {

  sub <- sens_long[sens_long$sens_type == tipo, ]

  modelo <- glmmTMB(n_sensillas ~ sexo * vista + flagelomero + (1 | individuo),
                    data   = sub,
                    family = poisson)

  modelos_glmm[[tipo]] <- modelo  # guardar para diagnóstico

  cat("\n============================================================\n")
  cat(sprintf("  GLMM - Tipo: %s\n", tipo))
  cat("============================================================\n")
  print(summary(modelo))
}

Diagnóstico de supuestos con DHARMa

Para cada tipo de sensilla se evalúa:

- Uniformidad de residuos (prueba KS): si p > 0.05 el modelo ajusta bien

- Sobredispersión: si p < 0.05 la varianza real supera la esperada bajo

Poisson y habría que cambiar a distribución binomial negativa

- Outliers: observaciones extremas influyentes

tabla_dharma <- data.frame(
  Tipo          = character(),
  KS_stat       = numeric(),
  KS_p          = numeric(),
  Dispersion_ratio = numeric(),
  Dispersion_p  = numeric(),
  Outliers_p    = numeric(),
  Conclusion    = character(),
  stringsAsFactors = FALSE
)

for (tipo in levels(sens_long$sens_type)) {

  cat("\n", rep("─", 55), "\n")
  cat("  DHARMa diagnóstico:", tipo, "\n")
  cat(rep("─", 55), "\n")

  mod     <- modelos_glmm[[tipo]]
  sim_res <- simulateResiduals(fittedModel = mod, n = 500, plot = FALSE)

  # Gráficos QQ y residuos vs predichos
  par(mfrow = c(1, 2))
  plotQQunif(sim_res,    main = paste("QQ-plot:", tipo))
  plotResiduals(sim_res, main = paste("Residuos vs Predichos:", tipo))
  par(mfrow = c(1, 1))

  # Tests formales
  test_unif <- testUniformity(sim_res,  plot = FALSE)
  test_disp <- testDispersion(sim_res,  plot = FALSE)
  test_out  <- testOutliers(sim_res,    plot = FALSE)

  cat("\n  >> Test de uniformidad (KS):\n")
  cat(sprintf("     D = %.4f,  p = %.4f\n", test_unif$statistic, test_unif$p.value))

  cat("\n  >> Test de sobredispersión:\n")
  cat(sprintf("     ratio = %.4f,  p = %.4f\n", test_disp$statistic, test_disp$p.value))

  cat("\n  >> Test de outliers:\n")
  cat(sprintf("     p = %.4f\n", test_out$p.value))

  conclusion <- ifelse(test_disp$p.value < 0.05,
                       "SOBREDISPERSIÓN → reajustar con Binomial Negativa",
                       "Sin sobredispersión → Poisson adecuado")
  cat("\n  >> Conclusión:", conclusion, "\n")

  tabla_dharma <- rbind(tabla_dharma, data.frame(
    Tipo             = tipo,
    KS_stat          = round(test_unif$statistic, 4),
    KS_p             = round(test_unif$p.value,   4),
    Dispersion_ratio = round(test_disp$statistic, 4),
    Dispersion_p     = round(test_disp$p.value,   4),
    Outliers_p       = round(test_out$p.value,    4),
    Conclusion       = conclusion
  ))
}
